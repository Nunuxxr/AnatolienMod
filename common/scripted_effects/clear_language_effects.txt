### Clears the language flags of a state
#
# How to use:
#
#	state = {
#		clear_language = yes
#	}
#
# Used for dynamic state names
#
#
clear_language = {
	hidden_effect = {
		if={
			limit={has_state_flag=armenian_name}
			clr_state_flag = armenian_name
		}
		else_if = {
			limit={has_state_flag=azerbaijani_name}
			clr_state_flag = azerbaijani_name
		}
	}
}
